Ejercicio 2

Realizar la siguiente figura por medio del ciclo Mientras - Hacer



                        package com.sofka.taller3;

                        public class Ejercicio2T3 {
                            private int i, j, k, control;
                            
                            public void Ejercicio2T3() {
                                control = 1;
                                
                                while (control == 1) {
                                    
                                    for (int i = 10; i >= 1; i--) {

                                        for (int j = 9; j >= (10 - i + 1); j--) {
                                            System.out.print(" ");
                                        }

                                        for (int k = 10; k >= i; k--) {
                                            System.out.print("*");
                                        }
                                        
                                        System.out.println(" ");
                                    }
                                    control = 2;
                                }
                            }
                        }

Ejercicio 4

Se requiere un programa que pueda crear la tabla de multiplicar del número que el usuario indique por medio del ciclo Para; esta debe ser impresa del 1 al 10. Ejemplo, si el usuario indica que desea crear la tabla del 5, entonces el resultado debería ser el siguiente:



                        package com.sofka.taller3;

                        import java.util.Scanner;

                        public class Ejercicio4T3 {
                            private int n, i, res;
                            private Scanner input;
                            
                            public void Ejercicio4T3() {
                                input = new Scanner (System.in);
                                
                                System.out.println("**Tablas de multiplicar");
                                
                                askN();
                                n = captureDataInt();
                                
                                for (int i = 1; i <= 10; i++) {
                                    res = (i * n);
                                    System.out.println(i + "*" + n + "=" + res);
                                }
                            }
                            
                            private void askN() {System.out.println("Indique número para crear tabla de multiplicar");}
                            
                            private int captureDataInt() {
                                return input.nextInt();
                            }
                        }

Ejercicio 5

Utilizando el ciclo que usted desee, crear un menú de ejecución infinita hasta que el usuario desee terminar dicho ciclo.



                        package com.sofka.taller3;

                        import java.io.IOException;
                        import java.util.Scanner;

                        public class Ejercicio5T3 {
                            private String name = "usuario";
                            private int option;
                            private Scanner input;
                            
                            public void Ejercicio5T3() throws IOException {
                                input = new Scanner (System.in);
                                
                                do {
                                    menu();
                                    switch (option) {
                                    case 1:
                                        askName();
                                        name = captureDataString();
                                        break;
                                        
                                    case 2:
                                        System.out.println("Hola " + name);
                                        break;
                                        
                                    case 3:
                                        break;

                                    default:
                                        System.out.println("Opción inválida");
                                        break;
                                    }
                                    System.out.println("Presione 'enter' para continuar");
                                    System.in.read();
                                } while (option != 3);
                            System.out.println("Vuelva pronto");
                            }
                            

                            public void menu() {
                                System.out.println("1. Capturar nombre");
                                System.out.println("2. Saludar persona");
                                System.out.println("3. Salir del sistema");
                                option = captureOption();
                            }
                            
                            private void askName() {System.out.println("Ingrese su nombre");}
                            
                            public int captureOption() {
                                Scanner input = new Scanner (System.in);
                                return input.nextInt();
                            }
                            
                            private String captureDataString() {
                                return input.nextLine();
                            }
                        }
